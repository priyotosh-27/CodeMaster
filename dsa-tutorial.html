Arrays

An Array is a fixed-size, contiguous memory block that stores multiple elements of the same data type. Each element is stored in a sequential memory location and can be accessed using an index (starting from 0). Arrays provide O(1) time complexity for element access, making them one of the most widely used data structures in programming.

  • Access: O(1) for any index
  • Search: O(n) in unsorted arrays, O(log n) in sorted arrays (binary search)
  • Insertion / Deletion: O(n) (because elements may need to be shifted)
  • Applications: Lookup tables, matrices, heaps, stacks, queues, buffers, image processing, and more
Example: Reverse an Array

Problem: Given an array, reverse the elements in-place.

Input: arr = [1, 2, 3, 4, 5]
Output: [5, 4, 3, 2, 1]

Explanation: Swap the first element with the last, second with second last, and so on.

// Java: Reverse an Array
int[] arr = {1, 2, 3, 4, 5};
int n = arr.length;
for (int i = 0; i < n / 2; i++) { // iterate halfway
    int temp = arr[i]; // store first element
    arr[i] = arr[n - i - 1]; // swap with last
    arr[n - i - 1] = temp; // place stored value
}
// Output: 5 4 3 2 1

// C++: Reverse an Array
int arr[] = {1, 2, 3, 4, 5};
int n = 5;
for (int i = 0; i < n / 2; i++) {
    int temp = arr[i];
    arr[i] = arr[n - i - 1];
    arr[n - i - 1] = temp;
}
// Output: 5 4 3 2 1

# Python: Reverse an Array
arr = [1, 2, 3, 4, 5]
arr.reverse()  # in-place reversal
# Output: [5, 4, 3, 2, 1]

1. Find the Maximum and Minimum in an Array

Problem: Given an array, find the largest and smallest numbers.

Input: [3, 1, 4, 1, 5, 9]
Output: Max: 9, Min: 1

Explanation: Iterate and compare each element to track max and min.

// Java
int[] arr = {3, 1, 4, 1, 5, 9};
int max = arr[0], min = arr[0];
for (int val : arr) { // loop through array
    if (val > max) max = val; // update max
    if (val < min) min = val; // update min
}
System.out.println("Max: " + max + ", Min: " + min);

// C++
int arr[] = {3, 1, 4, 1, 5, 9};
int n = 6;
int max = arr[0], min = arr[0];
for (int i = 0; i < n; i++) {
    if (arr[i] > max) max = arr[i];
    if (arr[i] < min) min = arr[i];
}
cout << "Max: " << max << ", Min: " << min;

# Python
arr = [3, 1, 4, 1, 5, 9]
print("Max:", max(arr), ", Min:", min(arr))
2. Rotate an Array by K Positions

Problem: Rotate array elements to the right by k steps.

Input: [1, 2, 3, 4, 5], k=2
Output: [4, 5, 1, 2, 3]

Explanation: Last k elements move to front, rest shift right.

// Java: Rotate right by k
int[] arr = {1, 2, 3, 4, 5};
int k = 2;
k = k % arr.length; // handle k > length
int[] rotated = new int[arr.length];
for (int i = 0; i < arr.length; i++) {
    rotated[(i + k) % arr.length] = arr[i]; // shift position
}
System.out.println(Arrays.toString(rotated));

// C++
int arr[] = {1, 2, 3, 4, 5};
int n = 5, k = 2;
k = k % n;
int rotated[5];
for (int i = 0; i < n; i++) {
    rotated[(i + k) % n] = arr[i];
}
for (int x : rotated) cout << x << " ";

# Python
arr = [1, 2, 3, 4, 5]
k = 2 % len(arr)
rotated = arr[-k:] + arr[:-k]  # slicing trick
print(rotated)
3. Find the Second Largest Element

Problem: Find the second largest unique number in the array.

Input: [10, 20, 4, 45, 99]
Output: 45

Explanation: Track largest and second largest while iterating.

// Java
int[] arr = {10, 20, 4, 45, 99};
int first = Integer.MIN_VALUE, second = Integer.MIN_VALUE;
for (int num : arr) {
    if (num > first) { // update both
        second = first;
        first = num;
    } else if (num > second && num != first) {
        second = num;
    }
}
System.out.println("Second Largest: " + second);

// C++
int arr[] = {10, 20, 4, 45, 99};
int n = 5;
int first = INT_MIN, second = INT_MIN;
for (int i = 0; i < n; i++) {
    if (arr[i] > first) {
        second = first;
        first = arr[i];
    } else if (arr[i] > second && arr[i] != first) {
        second = arr[i];
    }
}
cout << "Second Largest: " << second;

# Python
arr = [10, 20, 4, 45, 99]
unique_sorted = sorted(set(arr), reverse=True)
print("Second Largest:", unique_sorted[1])

Linked Lists

A Linked List is a linear data structure where elements (nodes) are connected using pointers. Each node contains data and a reference (or link) to the next (and/or previous) node. Linked lists are dynamic in size and allow efficient insertions/deletions.

Singly Linked List Doubly Linked List Circular Singly Circular Doubly

Singly Linked List

Each node contains data and a pointer to the next node. The last node points to NULL. 

[Head] → [Data|Next] → [Data|Next] → NULL
Operation Time Complexity
Insert at head O(1)
Insert at end O(n)
Search O(n) 
// Java - Insert at Head
class Node {
    int data;
    Node next;
    Node(int data) { this.data = data; }
}
class LinkedList {
    Node head;
    void insertAtHead(int data) {
        Node newNode = new Node(data);
        newNode.next = head;
        head = newNode;
    }
}

        Node* newNode = new Node(data);
        newNode->next = head;
        head = newNode;
    }
};

Doubly Linked List

Each node contains data, a pointer to the next node, and a pointer to the previous node. Allows traversal in both directions. 

NULL <- [Prev|Data|Next] <-> [Prev|Data|Next] <-> [Prev|Data|Next] -> NULL
Operation Time Complexity
Insert/Delete at head/tail O(1)
Search O(n)

Circular Singly Linked List

Like a singly linked list, but the last node points back to the head, forming a circle. No node points to NULL. 

[Head] → [Data|Next] → [Data|Next] → ... → [Data|Next] --+
        ^-----------------------------------------------+
Operation Time Complexity
Insert at head/tail O(1) (with tail pointer)
Search O(n) 
// Java - Insert at End (with tail pointer)
class Node {
    int data;
    Node next;
    Node(int data) { this.data = data; }
}
class CircularSinglyLinkedList {
    Node tail = null;
    void insertAtEnd(int data) {
        Node newNode = new Node(data);
        if (tail == null) {
            tail = newNode;
            tail.next = tail;
        } else {
            newNode.next = tail.next;
            tail.next = newNode;
            tail = newNode;
        }
    }
}

            tail->next = tail;

            tail->next = newNode;
            tail = newNode;
        }
    }
};

Stacks

A Stack is a linear data structure that follows the LIFO (Last-In-First-Out) principle. The last element inserted is the first one to be removed. Stacks are used for function call management, undo operations, expression evaluation, and more. 

Top
 ↑
[ 30 ] ← push/pop
[ 20 ]
[ 10 ] ← Bottom
Operation Time Complexity
Push O(1)
Pop O(1)
Peek/Top O(1)

Stack Using Array (from scratch)

// Java: Stack using Array
class Stack {
    int max = 1000;
    int[] arr = new int[max];
    int top = -1;
    void push(int x) {
        if (top == max - 1) System.out.println("Overflow");
        else arr[++top] = x;
    }
    int pop() {
        if (top == -1) { System.out.println("Underflow"); return -1; }
        return arr[top--];
    }
    int peek() {
        if (top == -1) { System.out.println("Empty"); return -1; }
        return arr[top];
    }
}

Queues

A Queue is a linear data structure that follows the FIFO (First-In-First-Out) principle. The first element inserted is the first one to be removed. Queues are used in scheduling, buffering, and breadth-first search. 

Front                Rear
 ↓                    ↓
[ 10 ] [ 20 ] [ 30 ] [   ]
Operation Time Complexity
Enqueue (push) O(1)
Dequeue (pop) O(1)
Peek (front) O(1)

Queue Using Array (from scratch)

// Java: Queue using Array (circular)
class Queue {
    int max = 1000;
    int[] arr = new int[max];
    int front = 0, rear = 0, size = 0;
    void enqueue(int x) {
        if (size == max) System.out.println("Overflow");
        else {
            arr[rear] = x;
            rear = (rear + 1) % max;
            size++;
        }
    }
    int dequeue() {
        if (size == 0) { System.out.println("Underflow"); return -1; }
        int val = arr[front];
        front = (front + 1) % max;
        size--;
        return val;
    }
    int peek() {
        if (size == 0) { System.out.println("Empty"); return -1; }
        return arr[front];
    }
}

class Queue:

        self.rear = 0
        self.size = 0
        self.max = max_size

        if self.size == self.max:
            print("Overflow")

            self.rear = (self.rear + 1) % self.max

        self.front = (self.front + 1) % self.max
        self.size -= 1
        return val
    def peek(self):
        if self.size == 0:
            print("Empty")
            return None
        return self.arr[self.front]

Binary Tree

A Binary Tree is a tree in which each node has at most two children (left and right). Used as the basis for many other tree structures. 

        1
       / \
      2   3
     / \

Operation Time Complexity

// Java: Inorder Traversal
void inorder(Node root) {
    if (root == null) return;
    inorder(root.left);
    System.out.print(root.data + " ");
    inorder(root.right);
}

void preorder(Node root) {
    if (root == null) return;

    preorder(root.left);
    preorder(root.right);
}

// Java: Postorder Traversal
void postorder(Node root) {
    if (root == null) return;
    postorder(root.left);

Dijkstra's Algorithm

Finds shortest path from a source to all nodes in a weighted graph with non-negative weights.

// Java: Dijkstra (adjacency list)

    int[] cur = pq.poll();

        int v = edge[0], w = edge[1];

        }
    }
}

Binary Search

Efficiently search a sorted array by repeatedly dividing the search interval in half.

// Java: Binary Search
int binarySearch(int[] arr, int target) {
    int l = 0, r = arr.length - 1;
    while (l <= r) {
        int m = l + (r - l) / 2;
        if (arr[m] == target) return m;
        if (arr[m] < target) l = m + 1;
        else r = m - 1;
    }
    return -1;
}

Jump Search

Searches sorted arrays by jumping ahead by fixed steps and then performing linear search in a block.

// Java: Jump Search
int jumpSearch(int[] arr, int target) {
    int n = arr.length;
    int step = (int)Math.sqrt(n);
    int prev = 0;
    while (arr[Math.min(step, n)-1] < target) {
        prev = step;
        step += (int)Math.sqrt(n);
        if (prev >= n) return -1;
    }
    for (int i = prev; i < Math.min(step, n); i++)
        if (arr[i] == target) return i;
    return -1;
}

Interpolation Search

Improves binary search for uniformly distributed sorted arrays by estimating the position.

// Java: Interpolation Search
int interpolationSearch(int[] arr, int target) {
    int low = 0, high = arr.length - 1;
    while (low <= high && target >= arr[low] && target <= arr[high]) {
        int pos = low + ((target - arr[low]) * (high - low)) / (arr[high] - arr[low]);
        if (arr[pos] == target) return pos;
        if (arr[pos] < target) low = pos + 1;
        else high = pos - 1;
    }
    return -1;
}

Ternary Search

Divides the array into three parts. Used for unimodal functions (single peak/minimum).

// Java: Ternary Search
int ternarySearch(int[] arr, int target) {
    int l = 0, r = arr.length - 1;
    while (l <= r) {
        int mid1 = l + (r - l) / 3;
        int mid2 = r - (r - l) / 3;
        if (arr[mid1] == target) return mid1;
        if (arr[mid2] == target) return mid2;
        if (target < arr[mid1]) r = mid1 - 1;
        else if (target > arr[mid2]) l = mid2 + 1;
        else { l = mid1 + 1; r = mid2 - 1; }
    }
    return -1;
}

Merge Sort

Divide and conquer algorithm that splits the array, sorts each half, and merges them.

// Java: Merge Sort
void mergeSort(int[] arr, int l, int r) {
    if (l < r) {
        int m = l + (r - l) / 2;
        mergeSort(arr, l, m);
        mergeSort(arr, m + 1, r);
        merge(arr, l, m, r);
    }
}
void merge(int[] arr, int l, int m, int r) {
    int n1 = m - l + 1, n2 = r - m;
    int[] L = new int[n1], R = new int[n2];
    for (int i = 0; i < n1; i++) L[i] = arr[l + i];
    for (int j = 0; j < n2; j++) R[j] = arr[m + 1 + j];
    int i = 0, j = 0, k = l;
    while (i < n1 && j < n2)
        arr[k++] = (L[i] <= R[j]) ? L[i++] : R[j++];
    while (i < n1) arr[k++] = L[i++];
    while (j < n2) arr[k++] = R[j++];
}

Quick Sort

Divide and conquer algorithm that picks a pivot and partitions the array around the pivot.

void quickSort(int[] arr, int low, int high) {
    if (low < high) {
        int pi = partition(arr, low, high);
        quickSort(arr, low, pi-1);
        quickSort(arr, pi+1, high);
    }

int partition(int[] arr, int low, int high) {
    int pivot = arr[high], i = low-1;
    for (int j=low; j< high; j++)
        if (arr[j] < pivot)
            swap(arr, ++i, j);
    swap(arr, i+1, high);
    return i+1;
}

    int t = arr[i]; arr[i] = arr[j]; arr[j] = t;
}

void heapify(int[] arr, int n, int i) {
    int largest = i, l = 2*i+1, r = 2*i+2;
    if (l < n && arr[l] > arr[largest]) largest = l;
    if (r < n && arr[r] > arr[largest]) largest = r;

        int t = arr[i]; arr[i] = arr[largest]; arr[largest] = t;

}

        heapify(arr, i, 0);
    }
}

Radix Sort

Sorts numbers digit by digit using a stable sort (like counting sort) as a subroutine.

void countingSort(int[] arr, int exp) {
    int n = arr.length;
    int[] output = new int[n], count = new int[10];
    for (int i = 0; i < n; i++) count[(arr[i]/exp)%10]++;
    for (int i = 1; i < 10; i++) count[i] += count[i-1];

}

Recursion

Recursion is a technique where a function calls itself to solve smaller instances of the same problem. Each recursive solution needs a base case (to stop recursion) and a recursive case. Recursion uses the call stack and is ideal for problems that can be broken into similar subproblems.

  • Base case: condition to stop recursion
  • Recursive case: reduces problem size and calls itself
  • Call stack: each recursive call pushes a frame (watch out for stack overflow)
  • When to use: tree traversals, divide & conquer, combinatorics, DP (top-down)
Example: Factorial (Recursive)

Problem: Compute n! using recursion.

Backtracking

Backtracking is a refinement of recursion used to build solutions incrementally and abandon (backtrack) when a partial solution cannot lead to a valid full solution. It's commonly used for constraint-satisfaction problems and combinatorial search.

  • Approach: try choices one-by-one, recurse, undo (backtrack) the choice
  • Use-cases: permutations, combinations, N-Queens, Sudoku, word search
  • Complexity: often exponential — prune with constraints where possible
Example: Generate Permutations (Backtracking)

// Java: Permutations (backtracking)
void permute(int[] arr, int l, List res) {
    if (l == arr.length) {
        List r = new ArrayList<>();
        for (int x : arr) r.add(x);
        res.add(r);
        return;
    }
    for (int i = l; i < arr.length; i++) {
        swap(arr, l, i);
        permute(arr, l+1, res);
        swap(arr, l, i); // backtrack
    }
}

Dynamic Programming (DP) is a method for solving problems by breaking them down into overlapping subproblems and storing the results of these subproblems to avoid redundant computations. It is mainly used for optimization problems where results of smaller subproblems are reused.

Key Points

  • Stores results of subproblems to avoid recalculating.
  • Two main types: Top-down (Memoization) and Bottom-up (Tabulation).
  • Helps in optimizing recursive algorithms.
  • Common in problems involving counting, paths, sequences, and optimization.

Greedy Algorithm

A Greedy Algorithm builds up a solution step-by-step, always choosing the next step that offers the most immediate benefit. It works best for problems where a global optimum can be reached through locally optimal steps.

Steps to Solve a Problem Using Greedy Algorithm

  • Identify if local optimal choices lead to a global optimal solution.
  • At each step, choose the option that looks best at the moment.
  • Repeat until the problem is solved.
  • Verify if the solution is indeed optimal.

Advanced Trees include specialized tree data structures and algorithms used to solve complex problems efficiently. These include AVL trees, Red-Black trees, Segment Trees, Fenwick Trees (Binary Indexed Trees), and Trie structures. They optimize dynamic data operations like searching, updating, and range queries.

Key Points

  • Balanced trees maintain height to optimize search and update times.
  • Segment Trees and Fenwick Trees enable efficient range queries and updates.
  • Tries are used for efficient string prefix searches.
  • Red-Black and AVL trees self-balance after insertions/deletions.

class SegmentTree {
    int[] tree;
    int n;

    SegmentTree(int[] arr) {
        n = arr.length;
        tree = new int[4 * n];
        build(arr, 1, 0, n - 1);
    }

    void build(int[] arr, int node, int start, int end) {
        if (start == end) {
            tree[node] = arr[start];
            return;
        }
        int mid = (start + end) / 2;
        build(arr, 2 * node, start, mid);
        build(arr, 2 * node + 1, mid + 1, end);
        tree[node] = tree[2 * node] + tree[2 * node + 1];
    }

    int query(int node, int start, int end, int l, int r) {
        if (r < start || l > end) return 0;
        if (l <= start && end <= r) return tree[node];
        int mid = (start + end) / 2;
        return query(2 * node, start, mid, l, r) + query(2 * node + 1, mid + 1, end, l, r);
    }
}

            children[i] = nullptr;
        isEndOfWord = false;

class Trie {
    TrieNode* root;
public:
    Trie() { root = new TrieNode(); }

    void insert(string key) {
        TrieNode* node = root;
        for (char c : key) {
            int index = c - 'a';
            if (!node->children[index])
                node->children[index] = new TrieNode();
            node = node->children[index];
        }
        node->isEndOfWord = true;
    }

    bool search(string key) {
        TrieNode* node = root;
        for (char c : key) {
            int index = c - 'a';
            if (!node->children[index])
                return false;
            node = node->children[index];
        }
        return node->isEndOfWord;
    }
};

D i s j o i n t   S e t   (Union-Find)

Disjoint Set (Union-Find) is a data structure that keeps track of a set of elements partitioned into disjoint (non-overlapping) subsets. It supports two useful operations efficiently: Find (which subset an element belongs to) and Union (join two subsets). Widely used in network connectivity, Kruskal’s MST algorithm, and more.

Key Operations

  • Find: Returns representative (root) of the set containing an element.
  • Union: Merges two subsets into a single subset.
  • Optimizations: Path compression and Union by rank/size improve efficiency.

Java Implementation


public class DisjointSet {
    private int[] parent, rank;

    public DisjointSet(int n) {
        parent = new int[n];
        rank = new int[n];
        for (int i = 0; i < n; i++) {
            parent[i] = i;
            rank[i] = 0;
        }
    }

    public int find(int x) {
        if (parent[x] != x)
            parent[x] = find(parent[x]);  // Path compression
        return parent[x];
    }

    public void union(int x, int y) {
        int rootX = find(x);
        int rootY = find(y);
        if (rootX == rootY) return;

        // Union by rank
        if (rank[rootX] < rank[rootY]) {
            parent[rootX] = rootY;
        } else if (rank[rootX] > rank[rootY]) {
            parent[rootY] = rootX;
        } else {
            parent[rootY] = rootX;
            rank[rootX]++;
        }
    }

    // Example usage
    public static void main(String[] args) {
        DisjointSet ds = new DisjointSet(5);
        ds.union(0, 2);
        ds.union(4, 2);
        ds.union(3, 1);

        System.out.println(ds.find(4) == ds.find(0));  // true
        System.out.println(ds.find(1) == ds.find(0));  // false
    }
}

H a s h i n g

Hashing is a technique to map data of arbitrary size to fixed-size values called hash codes, used for fast data retrieval. Hash tables provide average O(1) time complexity for insert, delete, and search operations. Common collision resolution techniques include chaining and open addressing.

Key Concepts

  • Hash Function: Converts input data to an index in hash table.
  • Collision: When two inputs map to the same index.
  • Collision Resolution: Techniques like chaining (linked lists) and open addressing (probing).
  • Load Factor: Ratio of number of elements to the table size, impacts performance.

Example: Simple Hash Table with Chaining (Java)


import java.util.LinkedList;

class HashNode {
    String key;
    int value;

    HashNode(String key, int value) {
        this.key = key;
        this.value = value;
    }
}

public class HashTable {
    private LinkedList[] table;
    private int capacity;

    @SuppressWarnings("unchecked")
    public HashTable(int capacity) {
        this.capacity = capacity;
        table = new LinkedList[capacity];
        for (int i = 0; i < capacity; i++) {
            table[i] = new LinkedList<>();
        }
    }

    private int hash(String key) {
        return Math.abs(key.hashCode()) % capacity;
    }

    public void put(String key, int value) {
        int index = hash(key);
        for (HashNode node : table[index]) {
            if (node.key.equals(key)) {
                node.value = value; // Update existing
                return;
            }
        }
        table[index].add(new HashNode(key, value)); // Insert new
    }

    public Integer get(String key) {
        int index = hash(key);
        for (HashNode node : table[index]) {
            if (node.key.equals(key)) {
                return node.value;
            }
        }
        return null; // Not found
    }

    public boolean remove(String key) {
        int index = hash(key);
        for (HashNode node : table[index]) {
            if (node.key.equals(key)) {
                table[index].remove(node);
                return true;
            }
        }
        return false; // Not found
    }

    public static void main(String[] args) {
        HashTable ht = new HashTable(10);
        ht.put("apple", 10);
        ht.put("banana", 20);
        ht.put("orange", 30);
        System.out.println("Value for apple: " + ht.get("apple"));
        ht.remove("banana");
        System.out.println("Value for banana: " + ht.get("banana")); // Should be null
    }
}

struct HashNode {
    string key;
    int value;
    HashNode(string k, int v) : key(k), value(v) {}
};

class HashTable {
    vector<list<HashNode>> table;
    int capacity;

    int hash(const string &key) {
        size_t hashVal = 0;
        for (char c : key)
            hashVal = 31 * hashVal + c;
        return hashVal % capacity;
    }

public:
    HashTable(int cap) : capacity(cap) {
        table.resize(capacity);
    }

    void put(const string &key, int value) {
        int index = hash(key);
        for (auto &node : table[index]) {
            if (node.key == key) {
                node.value = value; // Update
                return;
            }
        }
        table[index].emplace_back(key, value); // Insert
    }

    int* get(const string &key) {
        int index = hash(key);
        for (auto &node : table[index]) {
            if (node.key == key) {
                return &node.value;
            }
        }
        return nullptr; // Not found
    }

    bool remove(const string &key) {
        int index = hash(key);
        for (auto it = table[index].begin(); it != table[index].end(); ++it) {
            if (it->key == key) {
                table[index].erase(it);
                return true;
            }
        }
        return false; // Not found
    }
};
